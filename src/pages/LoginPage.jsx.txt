import React, { useState } from 'react';
import { Link, useNavigate } from 'react-router-dom';
import { Helmet } from 'react-helmet';
import { motion } from 'framer-motion';
import { useAuth } from '@/contexts/AuthContext.jsx';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import { Label } from '@/components/ui/label';
import { toast } from 'sonner';

const pageVariants = {
  initial: { opacity: 0 },
  animate: { opacity: 1, transition: { duration: 0.4 } },
  exit: { opacity: 0, transition: { duration: 0.3 } }
};

const LoginPage = () => {
  const navigate = useNavigate();
  const { login } = useAuth();
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const [loading, setLoading] = useState(false);

  const handleSubmit = async (e) => {
    e.preventDefault();
    setLoading(true);

    try {
      await login(email, password);
      toast('welcome back');
      navigate('/');
    } catch (error) {
      toast("something didn't feel right, try again");
    } finally {
      setLoading(false);
    }
  };

  return (
    <motion.div variants={pageVariants} initial="initial" animate="animate" exit="exit">
      <Helmet>
        <title>Login - Unsaid</title>
        <meta name="description" content="Login to access your private writings and echo mode" />
      </Helmet>

      <div className="min-h-screen flex items-center justify-center px-6 py-20">
        <div className="w-full max-w-md space-y-10">
          <div className="text-center">
            <h1 className="text-4xl md:text-5xl font-serif italic tracking-wide text-foreground">
              Welcome back
            </h1>
            <p className="mt-4 text-secondary-text/70 font-light">
              continue your journey of unspoken thoughts
            </p>
          </div>

          <form onSubmit={handleSubmit} className="glass-morphism p-10 rounded-3xl space-y-8">
            <div className="space-y-6">
              <div className="space-y-2">
                <Label htmlFor="email" className="text-sm text-secondary-text font-light tracking-wide">Email</Label>
                <Input
                  id="email"
                  type="email"
                  value={email}
                  onChange={(e) => setEmail(e.target.value)}
                  required
                  className="bg-white/5 border-white/10 text-foreground placeholder:text-secondary-text/30 focus:border-primary/50 focus:ring-primary/50 transition-all duration-300 rounded-xl py-6"
                  placeholder="your@email.com"
                />
              </div>

              <div className="space-y-2">
                <Label htmlFor="password" className="text-sm text-secondary-text font-light tracking-wide">Password</Label>
                <Input
                  id="password"
                  type="password"
                  value={password}
                  onChange={(e) => setPassword(e.target.value)}
                  required
                  className="bg-white/5 border-white/10 text-foreground placeholder:text-secondary-text/30 focus:border-primary/50 focus:ring-primary/50 transition-all duration-300 rounded-xl py-6"
                  placeholder="••••••••"
                />
              </div>
            </div>

            <Button
              type="submit"
              disabled={loading}
              className="w-full bg-primary/80 text-primary-foreground hover:bg-primary hover:shadow-[0_0_20px_rgba(139,123,168,0.4)] transition-all duration-300 rounded-full py-6 text-lg font-light tracking-wide"
            >
              {loading ? 'gathering thoughts...' : 'enter'}
            </Button>

            <p className="text-center text-sm text-secondary-text/60 font-light">
              don't have an account?{' '}
              <Link to="/signup" className="text-primary/80 hover:text-primary hover:underline transition-colors">
                sign up
              </Link>
            </p>
          </form>
        </div>
      </div>
    </motion.div>
  );
};

export default LoginPage;
