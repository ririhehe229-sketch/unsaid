import React, { useState, useEffect } from 'react';
import { Link, useNavigate } from 'react-router-dom';
import { Helmet } from 'react-helmet';
import { motion } from 'framer-motion';
import pb from '@/lib/pocketbaseClient';
import { useAuth } from '@/contexts/AuthContext.jsx';
import { Button } from '@/components/ui/button';
import { Badge } from '@/components/ui/badge';
import { toast } from 'sonner';

const pageVariants = {
  initial: { opacity: 0 },
  animate: { opacity: 1, transition: { duration: 0.4 } },
  exit: { opacity: 0, transition: { duration: 0.3 } }
};

const containerVariants = {
  hidden: { opacity: 0 },
  show: {
    opacity: 1,
    transition: { staggerChildren: 0.15 }
  }
};

const itemVariants = {
  hidden: { opacity: 0, y: 20 },
  show: { opacity: 1, y: 0, transition: { duration: 0.6, ease: "easeOut" } }
};

const HeldThoughtsPage = () => {
  const { currentUser } = useAuth();
  const navigate = useNavigate();
  const [drafts, setDrafts] = useState([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetchDrafts();
  }, []);

  const fetchDrafts = async () => {
    try {
      const records = await pb.collection('drafts').getList(1, 50, {
        filter: `user_id="${currentUser.id}"`,
        sort: '-created_at',
        $autoCancel: false
      });
      setDrafts(records.items);
    } catch (error) {
      console.error('Error fetching drafts:', error);
    } finally {
      setLoading(false);
    }
  };

  const handlePost = async (draft) => {
    try {
      const record = await pb.collection('thoughts').create({
        content: draft.content,
        emotion_tags: draft.emotion_tags,
        user_id: currentUser.id
      }, { $autoCancel: false });
      
      await pb.collection('drafts').delete(draft.id, { $autoCancel: false });
      toast("it's here now");
      navigate(`/thought/${record.id}`);
    } catch (error) {
      toast("something didn't feel right, try again");
    }
  };

  const handleDelete = async (id) => {
    try {
      await pb.collection('drafts').delete(id, { $autoCancel: false });
      setDrafts(drafts.filter(d => d.id !== id));
      toast('let it go');
    } catch (error) {
      toast("something didn't feel right, try again");
    }
  };

  if (loading) {
    return (
      <motion.div variants={pageVariants} initial="initial" animate="animate" exit="exit" className="min-h-screen flex items-center justify-center">
        <div className="text-2xl font-serif italic text-secondary-text animate-pulse tracking-wide">
          gathering thoughts...
        </div>
      </motion.div>
    );
  }

  return (
    <motion.div variants={pageVariants} initial="initial" animate="animate" exit="exit">
      <Helmet>
        <title>Held Thoughts - Unsaid</title>
        <meta name="description" content="Thoughts you're holding onto" />
      </Helmet>

      <div className="min-h-screen py-20 px-6 sm:px-8">
        <div className="max-w-3xl mx-auto">
          <motion.div 
            initial={{ opacity: 0, y: 20 }}
            animate={{ opacity: 1, y: 0 }}
            transition={{ duration: 1 }}
            className="text-center mb-16"
          >
            <h1 className="text-4xl md:text-5xl font-serif italic tracking-wide leading-tight text-foreground">
              Thoughts you held onto
            </h1>
            <p className="mt-4 text-lg text-secondary-text/70 font-light">
              waiting for the right moment
            </p>
          </motion.div>

          {drafts.length === 0 ? (
            <div className="text-center py-20">
              <p className="text-secondary-text text-lg mb-4 font-light">you aren't holding onto anything right now.</p>
              <Link to="/write">
                <Button variant="outline" className="border-white/10 text-secondary-text hover:bg-white/5 rounded-full px-6">
                  write something
                </Button>
              </Link>
            </div>
          ) : (
            <motion.div 
              variants={containerVariants}
              initial="hidden"
              animate="show"
              className="space-y-8"
            >
              {drafts.map((draft) => (
                <motion.div
                  key={draft.id}
                  variants={itemVariants}
                  className="glass-morphism p-8 rounded-2xl transition-all duration-500 hover:shadow-[0_10px_30px_rgba(0,0,0,0.2)] hover:border-white/10 animate-floating"
                >
                  <p className="text-xl leading-relaxed text-foreground/90 mb-6 font-serif italic">
                    "{draft.content}"
                  </p>
                  
                  {draft.emotion_tags && draft.emotion_tags.length > 0 && (
                    <div className="flex flex-wrap gap-3 mb-6">
                      {draft.emotion_tags.map((tag, i) => (
                        <Badge
                          key={i}
                          variant="secondary"
                          className="bg-white/5 text-secondary-text border-white/10 text-xs font-light tracking-wide px-3 py-1"
                        >
                          {tag}
                        </Badge>
                      ))}
                    </div>
                  )}
                  
                  <div className="flex flex-col sm:flex-row items-center justify-between gap-4 pt-4 border-t border-white/5">
                    <span className="text-xs text-secondary-text/40 font-light tracking-wider">
                      held since {new Date(draft.created_at).toLocaleDateString('en-US', { month: 'short', day: 'numeric' })}
                    </span>
                    
                    <div className="flex items-center gap-3 w-full sm:w-auto">
                      <Button
                        onClick={() => handlePost(draft)}
                        size="sm"
                        className="bg-primary/20 text-primary-foreground hover:bg-primary/80 transition-all duration-300 rounded-full px-4 font-light border border-primary/30 flex-1 sm:flex-none"
                      >
                        post it now
                      </Button>
                      <Button
                        onClick={() => navigate('/write', { state: { draft } })}
                        variant="outline"
                        size="sm"
                        className="border-white/10 text-secondary-text hover:bg-white/5 transition-all duration-300 rounded-full px-4 font-light flex-1 sm:flex-none"
                      >
                        edit
                      </Button>
                      <Button
                        onClick={() => handleDelete(draft.id)}
                        variant="ghost"
                        size="sm"
                        className="text-secondary-text/60 hover:text-destructive/80 hover:bg-destructive/10 transition-all duration-300 rounded-full px-4 font-light flex-1 sm:flex-none"
                      >
                        let it go
                      </Button>
                    </div>
                  </div>
                </motion.div>
              ))}
            </motion.div>
          )}
        </div>
      </div>
    </motion.div>
  );
};

export default HeldThoughtsPage;
