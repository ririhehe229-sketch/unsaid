import React, { useState } from 'react';
import { Link, useNavigate } from 'react-router-dom';
import { Helmet } from 'react-helmet';
import { motion } from 'framer-motion';
import { useAuth } from '@/contexts/AuthContext.jsx';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import { Label } from '@/components/ui/label';
import { toast } from 'sonner';

const pageVariants = {
  initial: { opacity: 0 },
  animate: { opacity: 1, transition: { duration: 0.4 } },
  exit: { opacity: 0, transition: { duration: 0.3 } }
};

const SignupPage = () => {
  const navigate = useNavigate();
  const { signup } = useAuth();
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const [passwordConfirm, setPasswordConfirm] = useState('');
  const [loading, setLoading] = useState(false);

  const handleSubmit = async (e) => {
    e.preventDefault();

    if (password !== passwordConfirm) {
      toast("passwords don't match");
      return;
    }

    if (password.length < 8) {
      toast('password must be at least 8 characters');
      return;
    }

    setLoading(true);

    try {
      await signup(email, password, passwordConfirm);
      toast("it's here now");
      navigate('/');
    } catch (error) {
      toast("something didn't feel right, try again");
    } finally {
      setLoading(false);
    }
  };

  return (
    <motion.div variants={pageVariants} initial="initial" animate="animate" exit="exit">
      <Helmet>
        <title>Sign Up - Unsaid</title>
        <meta name="description" content="Create an account to save private writings and discover echoes" />
      </Helmet>

      <div className="min-h-screen flex items-center justify-center px-6 py-20">
        <div className="w-full max-w-md space-y-10">
          <div className="text-center">
            <h1 className="text-4xl md:text-5xl font-serif italic tracking-wide text-foreground">
              Begin your journey
            </h1>
            <p className="mt-4 text-secondary-text/70 font-light">
              a space for thoughts you can't quite say
            </p>
          </div>

          <form onSubmit={handleSubmit} className="glass-morphism p-10 rounded-3xl space-y-8">
            <div className="space-y-6">
              <div className="space-y-2">
                <Label htmlFor="email" className="text-sm text-secondary-text font-light tracking-wide">Email</Label>
                <Input
                  id="email"
                  type="email"
                  value={email}
                  onChange={(e) => setEmail(e.target.value)}
                  required
                  className="bg-white/5 border-white/10 text-foreground placeholder:text-secondary-text/30 focus:border-primary/50 focus:ring-primary/50 transition-all duration-300 rounded-xl py-6"
                  placeholder="your@email.com"
                />
              </div>

              <div className="space-y-2">
                <Label htmlFor="password" className="text-sm text-secondary-text font-light tracking-wide">Password</Label>
                <Input
                  id="password"
                  type="password"
                  value={password}
                  onChange={(e) => setPassword(e.target.value)}
                  required
                  className="bg-white/5 border-white/10 text-foreground placeholder:text-secondary-text/30 focus:border-primary/50 focus:ring-primary/50 transition-all duration-300 rounded-xl py-6"
                  placeholder="••••••••"
                />
              </div>

              <div className="space-y-2">
                <Label htmlFor="passwordConfirm" className="text-sm text-secondary-text font-light tracking-wide">Confirm Password</Label>
                <Input
                  id="passwordConfirm"
                  type="password"
                  value={passwordConfirm}
                  onChange={(e) => setPasswordConfirm(e.target.value)}
                  required
                  className="bg-white/5 border-white/10 text-foreground placeholder:text-secondary-text/30 focus:border-primary/50 focus:ring-primary/50 transition-all duration-300 rounded-xl py-6"
                  placeholder="••••••••"
                />
              </div>
            </div>

            <Button
              type="submit"
              disabled={loading}
              className="w-full bg-primary/80 text-primary-foreground hover:bg-primary hover:shadow-[0_0_20px_rgba(139,123,168,0.4)] transition-all duration-300 rounded-full py-6 text-lg font-light tracking-wide"
            >
              {loading ? 'gathering thoughts...' : 'join'}
            </Button>

            <p className="text-center text-sm text-secondary-text/60 font-light">
              already have an account?{' '}
              <Link to="/login" className="text-primary/80 hover:text-primary hover:underline transition-colors">
                login
              </Link>
            </p>
          </form>
        </div>
      </div>
    </motion.div>
  );
};

export default SignupPage;
