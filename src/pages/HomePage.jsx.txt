import React, { useState, useEffect } from 'react';
import { Link } from 'react-router-dom';
import { Helmet } from 'react-helmet';
import { motion } from 'framer-motion';
import pb from '@/lib/pocketbaseClient';
import { Button } from '@/components/ui/button';
import { Badge } from '@/components/ui/badge';
import { Heart } from 'lucide-react';
import ExampleThoughts from '@/components/ExampleThoughts.jsx';
import WelcomeMessage from '@/components/WelcomeMessage.jsx';
import AboutModal from '@/components/AboutModal.jsx';

const pageVariants = {
  initial: { opacity: 0 },
  animate: { opacity: 1, transition: { duration: 0.4 } },
  exit: { opacity: 0, transition: { duration: 0.3 } }
};

const HomePage = () => {
  const [thoughts, setThoughts] = useState([]);
  const [loading, setLoading] = useState(true);
  const [reactionCounts, setReactionCounts] = useState({});
  const [isAboutOpen, setIsAboutOpen] = useState(false);

  useEffect(() => {
    fetchThoughts();
  }, []);

  const fetchThoughts = async () => {
    try {
      const records = await pb.collection('thoughts').getList(1, 50, {
        sort: '-created_at',
        $autoCancel: false
      });
      setThoughts(records.items);

      const counts = {};
      for (const thought of records.items) {
        const reactions = await pb.collection('reactions').getFullList({
          filter: `thought_id="${thought.id}"`,
          $autoCancel: false
        });
        counts[thought.id] = reactions.length;
      }
      setReactionCounts(counts);
    } catch (error) {
      console.error('Error fetching thoughts:', error);
    } finally {
      setLoading(false);
    }
  };

  const containerVariants = {
    hidden: { opacity: 0 },
    show: {
      opacity: 1,
      transition: { staggerChildren: 0.15 }
    }
  };

  const itemVariants = {
    hidden: { opacity: 0, y: 30 },
    show: { opacity: 1, y: 0, transition: { duration: 0.8, ease: "easeOut" } }
  };

  if (loading) {
    return (
      <motion.div variants={pageVariants} initial="initial" animate="animate" exit="exit" className="min-h-screen flex items-center justify-center">
        <div className="text-center">
          <div className="text-2xl font-serif italic text-secondary-text animate-pulse tracking-wide">
            gathering thoughts...
          </div>
        </div>
      </motion.div>
    );
  }

  return (
    <motion.div variants={pageVariants} initial="initial" animate="animate" exit="exit">
      <Helmet>
        <title>Unsaid - A place for incomplete thoughts</title>
        <meta name="description" content="Share and complete thoughts you can't quite express" />
      </Helmet>

      <WelcomeMessage />
      <AboutModal isOpen={isAboutOpen} onClose={() => setIsAboutOpen(false)} />

      <div className="min-h-screen py-20 px-6 sm:px-8 flex flex-col">
        <div className="max-w-3xl mx-auto w-full flex-grow">
          <motion.div 
            initial={{ opacity: 0, y: 20 }}
            animate={{ opacity: 1, y: 0 }}
            transition={{ duration: 1 }}
            className="text-center mb-20 space-y-8"
          >
            <h1 className="text-5xl md:text-7xl font-serif italic tracking-wide leading-tight text-foreground drop-shadow-sm">
              Incomplete thoughts,<br />waiting to be finished
            </h1>
            <p className="text-xl text-secondary-text/80 max-w-2xl mx-auto leading-relaxed font-light">
              A space for the things you can't quite say.
            </p>
            <div className="pt-4">
              <Link to="/write">
                <Button className="bg-primary/80 text-primary-foreground hover:bg-primary hover:shadow-[0_0_20px_rgba(139,123,168,0.5)] transition-all duration-500 rounded-full px-8 py-6 text-lg font-light tracking-wide">
                  leave it here
                </Button>
              </Link>
            </div>
          </motion.div>

          <div className="space-y-8">
            {thoughts.length === 0 ? (
              <ExampleThoughts />
            ) : (
              <motion.div 
                variants={containerVariants}
                initial="hidden"
                animate="show"
                className="space-y-8"
              >
                {thoughts.map((thought) => (
                  <motion.div key={thought.id} variants={itemVariants}>
                    <Link to={`/thought/${thought.id}`} className="block group">
                      <div className="glass-morphism p-8 rounded-2xl transition-all duration-500 group-hover:-translate-y-2 group-hover:shadow-[0_15px_40px_rgba(0,0,0,0.3)] group-hover:border-primary/30 relative overflow-hidden animate-floating">
                        <div className="absolute inset-0 bg-gradient-to-br from-primary/5 to-transparent opacity-0 group-hover:opacity-100 transition-opacity duration-500" />
                        
                        <p className="text-2xl leading-relaxed text-foreground/95 mb-6 font-serif relative z-10">
                          {thought.content}
                        </p>
                        
                        {thought.emotion_tags && thought.emotion_tags.length > 0 && (
                          <div className="flex flex-wrap gap-3 mb-5 relative z-10">
                            {thought.emotion_tags.map((tag, i) => (
                              <Badge
                                key={i}
                                variant="secondary"
                                className="bg-white/5 text-secondary-text border-white/10 text-xs font-light tracking-wide px-3 py-1 group-hover:bg-primary/10 group-hover:text-primary-foreground transition-colors duration-300"
                              >
                                {tag}
                              </Badge>
                            ))}
                          </div>
                        )}
                        
                        <div className="flex items-center text-sm text-secondary-text/60 font-light relative z-10 group-hover:text-primary/80 transition-colors duration-300">
                          <Heart size={14} className="mr-2" />
                          <span>{reactionCounts[thought.id] || 0}</span>
                        </div>
                      </div>
                    </Link>
                  </motion.div>
                ))}
              </motion.div>
            )}
          </div>
          
          {/* Hidden trigger text at the bottom */}
          <motion.div 
            initial={{ opacity: 0 }}
            animate={{ opacity: 1 }}
            transition={{ duration: 1, delay: 1.5 }}
            className="mt-32 pb-12 text-center"
          >
            <button
              onClick={() => setIsAboutOpen(true)}
              className="text-[13px] text-[#a0a0a0] font-serif italic hover:text-[#e8e8e8] transition-colors duration-500 tracking-wider focus:outline-none focus-visible:ring-1 focus-visible:ring-primary/50 rounded px-2 py-1"
            >
              why does this exist?
            </button>
          </motion.div>
        </div>
      </div>
    </motion.div>
  );
};

export default HomePage;
